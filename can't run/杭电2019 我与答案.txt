int insertdata(int a[],int n,int data)
{
    int i,j;
    for(i=0;i<n;i++)
    {
        if(data<a[0])
        {
            for(j=n-1;j>=0;j--)
            {
                a[j+1]=a[j];
            }
            a[0]=data;
            n++;
        }
        if(data>a[n-1])
        {
            a[n]=data;
            n++;
        }
        if(data>a[i]&&data<a[i+1])
        {
            for(j=n-1;j>i;j--)
            {
                a[j+1]=a[j];
            }
            a[i+1]=data;
            n++;
        }
    }
    return n;
}
