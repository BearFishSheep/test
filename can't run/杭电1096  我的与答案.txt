#include <stdio.h>
int main()
{
    int N,M,a,i,j,sum;
    while(scanf("%d",&N)!=EOF)
    {
        for(j=1;j<=N;j++)
        {
            scanf("%d",&M);
            sum=0;
            for(i=1;i<=M;i++)
            {
                scanf("%d",&a);
                sum+=a;
            }
            printf("%d\n\n",sum);
        }
    }
    return 0;
}



//正确做法
#include<stdio.h>
int main()
{
    int n,m,i,j,num,sum,temp=0;
    while(scanf("%d",&n)!=EOF)
    {
        for(i=0;i<n;i++)
        {
            sum=0;
            scanf("%d",&m);
            for(j=0;j<m;j++)
            {
                scanf("%d",&num);
                sum+=num;
            }
            temp++;
            if(temp==n)
                printf("%d\n",sum);
            else
                printf("%d\n\n",sum);
            }
            }
            return 0;
}
